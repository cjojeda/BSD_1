  end while;

create procedure pa_generar_dos_aleatorios(
  out valor1 int,
  out valor2 int)
begin
  etiqueta1:loop
    set valor1=rand()*10;
    set valor2=rand()*10;
    if valor1<>valor2 then
      leave etiqueta1;
    end if;
  end loop;
end //
